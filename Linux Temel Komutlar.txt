pwd = nerede olduğunu gösterir(dosya dizini)
ls  = dosyanın içinde ne var gösterir
ls -l = detaylı gösterir
man ls = ls ile ilgili olanları gösterir(man cd vb..)
ls -F = dosya isimlerini / ile ayırırls
cd / = köke iner
cd = açar
cd .. = geri gider
 | =iki komutu birleştirir
grep ssl = ssl ismindeki dosyaları filtreler
touch d1=d1 adında dosya oluşturur
cat(more) d1= d1 adındaki dosyanın içindekileri yazar
ctrl + c = komutlardan çıkar
mkdir k1=k1 adında klasör oluşturur
rm d1 = dosyayı siler
rm -r k1 = klasörü siler(dolu-boş)
rmdir k1 =boş klasörü siler
echo yankı > deneme = deneme dosyası oluşturur yankı yazar
echo $HOME = Home'un yerini gösterir
history = kullandığın bütün komutları gösterir
history > kayıt = history kayıt adındaki dosyaya kaydeder.
cp dosya klasör = desktoptaki dosyayı klasöre kopyaladı
mv dosya klasör = desktoptaki dosyayı klasöre kesip yapıştırır
tar -cvf dosyalar.tar d1 d2 d3 = sıkıştırır(zip)
tar -tvf dosyalar.tar = içinde ne var gösterir
tar -xvf dosyalar.tar = zipi açar
